Modal Alert

		
// Show Alert (equal Show Success)
O_o.alert('Simple message');

// Show Success
O_o.alert('Simple message', 'success');

// Show Warning
O_o.alert('Simple message', 'warning');

// Show Danger
O_o.alert('Simple message', 'danger');

// Show Info
O_o.alert('Simple message', 'info');

// Show Primary
O_o.alert('Simple message', 'primary');

// Show Default
O_o.alert('Simple message', 'default');

// Show Link
O_o.alert('Simple message', 'link');

Modal Conrifrm

		
// Show Confirm
O_o.confirm('Simple message',
	// Callback for accept result
	function(){
		alert('true');
	},

	// Callback for reject result
	function(){
		alert('false');
	}
);

Modal Prompt

		
// Show Prompt
O_o.prompt(
	// Text for window
	'This is prompt message',

	// Url action for form
	'/form_action/',

	// Name field
	'field_name',

	// Callback on submit form (optional)
	function(form){
		alert(form.find('.prompt_area').val());
		return false;
	}
);
