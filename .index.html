Modals

Modal Alert

		
// Show Alert (equal Show Success)
O_o.alert('Simple message');

// Show Success
O_o.alert('Simple message', 'success');

// Show Warning
O_o.alert('Simple message', 'warning');

// Show Danger
O_o.alert('Simple message', 'danger');

// Show Info
O_o.alert('Simple message', 'info');

// Show Primary
O_o.alert('Simple message', 'primary');

// Show Default
O_o.alert('Simple message', 'default');

// Show Link
O_o.alert('Simple message', 'link');
